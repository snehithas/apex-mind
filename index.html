        function handleTimeLogic() {

            const now = new Date();

            const today = now.toDateString();

            const yesterday = new Date(now);

            yesterday.setDate(yesterday.getDate() - 1);

            const yesterdayStr = yesterday.toDateString();


            if (state.lastReset !== today) {

                if (state.lastDate === yesterdayStr) {

                    state.streak += 1;

                } else if (state.lastDate !== today) {

                    state.streak = 0;

                }

                

                state.dailyCount = 0;

                state.lastReset = today;

                save();

            }

        }

        function gainXP(amt, type, value, target = null) {

        }

        function handleImport(event) {

            const file = event.target.files[0];

            if (!file) return;


            const reader = new FileReader();

            reader.onload = function(e) {

                try {

                    const importedState = JSON.parse(e.target.result);

                    if (importedState.xp !== undefined) {

                        state = importedState;

                        save();

                        updateUI();

                        showToast("SYSTEM SYNC SUCCESSFUL");

                        closeModal();

                    }

                } catch (err) {

                    showToast("ERROR: INVALID FILE");

                }

            };

            reader.readAsText(file);

        }

        let timerInterval, startTime, timerRunning = false;


        function toggleRecoveryTimer() {

            const btn = document.getElementById('timer-toggle');

            const display = document.getElementById('timer-display');

            if (!timerRunning) {

                startTime = new Date();

                timerRunning = true;

                btn.innerText = "ACTIVE";

                timerInterval = setInterval(() => {

                    const diff = Math.floor((new Date() - startTime) / 1000);

                    display.innerText = `${Math.floor(diff/60).toString().padStart(2,'0')}:${(diff%60).toString().padStart(2,'0')}`;

                }, 1000);

            } else {

                clearInterval(timerInterval);

                timerRunning = false;

                const total = Math.floor((new Date() - startTime) / 1000);

                gainXP(100, 'Recovery', `${Math.floor(total/60)}m ${total%60}s`);

                closeModal();

            }

        }
